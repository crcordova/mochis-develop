```text
# [slice or phase name]

status: closed|blocked

summary:
- [one sentence]

- [objective and authorized boundary]

durable_decisions:
- [decision or none]

- [evidence note or none]

- read: [paths or none]
- changed: [paths or none]

- [performed, skipped, or not authorized]

blockers:
- [resolved, unresolved, or none]

- closure_ready: yes|no
- [review outcome or none]

next_recommendation:
- [next slice, handoff, or none]
```
